Intro to using AtlasCloud Alpine package repo

Install gpg key

curl -o /etc/apk/keys/packages@atlascloud.xyz-5c7e1f40.rsa.pub packages.atlascloud.xyz/atlascloud/alpine/packages@atlascloud.xyz-5c7e1f40.rsa.pub

add repo

echo https://packages.atlascloud.xyz/atlascloud/alpine/3.12/main >> /etc/apk/repositories

update

apk update

install stuff

apk add linux-vanilla-atlas

$

echo profit
